CREATE TABLE cozinheiro(
	id SERIAL NOT NULL,
	email CHAR(140),
	senha CHAR(16),
	nome CHAR(64),
	data_nasc DATE,
	cpf CHAR(11),
	especializacao CHAR(24),
	experiencia INT,
	receitas INT,
	contato CHAR(140),
	primary key(id)
);

CREATE TABLE receita(
	id SERIAL NOT NULL,
	nome char(64),
	descricao char(240),
	primary key(id)
);

CREATE TABLE especializacao(
	id SERIAL NOT NULL,
	nome char(64),
	nivel int,
	primary key(id)
);

ALTER TABLE receita
	ADD COLUMN id_cozinheiro INT,
	ADD CONSTRAINT fk_cozinheiro
	FOREIGN KEY (id_cozinheiro) REFERENCES cozinheiro(id);
